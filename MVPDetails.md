## 2. Database Schema — Drizzle

**File: `src/db/schema.ts`**

```typescript
import {
  pgTable, uuid, text, timestamp, real,
  boolean, integer, pgEnum, uniqueIndex
} from 'drizzle-orm/pg-core'

// Better Auth creates its own tables (user, session, account, verification)
// Import the user table from Better Auth's schema or reference by string
// We reference userId as text since Better Auth uses its own ID format

export const invoiceStatusEnum = pgEnum('invoice_status', [
  'draft', 'sent', 'paid', 'overdue', 'cancelled'
])

export const recurringIntervalEnum = pgEnum('recurring_interval', [
  'weekly', 'monthly', 'quarterly'
])

export const businesses = pgTable('businesses', {
  id:               uuid('id').primaryKey().defaultRandom(),
  userId:           text('user_id').notNull().unique(), // FK to Better Auth user
  name:             text('name').notNull(),
  email:            text('email').notNull(),
  phone:            text('phone'),
  address:          text('address'),
  logoUrl:          text('logo_url'),
  taxNumber:        text('tax_number'),
  defaultTaxRate:   real('default_tax_rate').default(0).notNull(),
  bankDetails:      text('bank_details'),
  currency:         text('currency').default('GBP').notNull(),
  createdAt:        timestamp('created_at').defaultNow().notNull(),
})

export const clients = pgTable('clients', {
  id:        uuid('id').primaryKey().defaultRandom(),
  userId:    text('user_id').notNull(),
  name:      text('name').notNull(),
  email:     text('email').notNull(),
  phone:     text('phone'),
  address:   text('address'),
  createdAt: timestamp('created_at').defaultNow().notNull(),
})

export const invoices = pgTable('invoices', {
  id:                uuid('id').primaryKey().defaultRandom(),
  userId:            text('user_id').notNull(),
  clientId:          uuid('client_id').notNull().references(() => clients.id),
  invoiceNumber:     text('invoice_number').notNull(),
  status:            invoiceStatusEnum('status').default('draft').notNull(),
  issueDate:         timestamp('issue_date').defaultNow().notNull(),
  dueDate:           timestamp('due_date').notNull(),
  notes:             text('notes'),
  taxRate:           real('tax_rate').default(0).notNull(),
  subtotal:          real('subtotal').default(0).notNull(),
  taxAmount:         real('tax_amount').default(0).notNull(),
  total:             real('total').default(0).notNull(),
  sentAt:            timestamp('sent_at'),
  paidAt:            timestamp('paid_at'),
  remindersSent:     integer('reminders_sent').default(0).notNull(),
  isRecurring:       boolean('is_recurring').default(false).notNull(),
  recurringInterval: recurringIntervalEnum('recurring_interval'),
  nextInvoiceDate:   timestamp('next_invoice_date'),
  createdAt:         timestamp('created_at').defaultNow().notNull(),
  updatedAt:         timestamp('updated_at').defaultNow().notNull(),
}, (table) => ({
  invoiceNumberIdx: uniqueIndex('invoice_number_idx').on(table.invoiceNumber),
}))

export const lineItems = pgTable('line_items', {
  id:          uuid('id').primaryKey().defaultRandom(),
  invoiceId:   uuid('invoice_id').notNull().references(() => invoices.id, { onDelete: 'cascade' }),
  description: text('description').notNull(),
  quantity:    real('quantity').notNull(),
  unitPrice:   real('unit_price').notNull(),
  amount:      real('amount').notNull(), // quantity * unitPrice, stored
})

// Type exports — use these everywhere instead of raw query return types
export type Business     = typeof businesses.$inferSelect
export type Client       = typeof clients.$inferSelect
export type Invoice      = typeof invoices.$inferSelect
export type LineItem     = typeof lineItems.$inferSelect
export type NewInvoice   = typeof invoices.$inferInsert
export type NewLineItem  = typeof lineItems.$inferInsert
```

### Getting the current user in a Server Action

Every server action that touches the database needs to verify the session first. Make a reusable helper:

**File: `src/lib/get-session.ts`**

```typescript
import { auth } from '@/lib/auth'
import { headers } from 'next/headers'

export async function getSession() {
  const session = await auth.api.getSession({
    headers: await headers(),
  })
  if (!session?.user) return null
  return session
}

// Use this in actions where auth is required
export async function requireAuth() {
  const session = await getSession()
  if (!session) throw new Error('UNAUTHORIZED')
  return session
}
```

**Usage in every server action:**
```typescript
'use server'
import { requireAuth } from '@/lib/get-session'

export async function someAction() {
  const session = await requireAuth() // throws if not logged in
  const userId = session.user.id
  // ... rest of action
}
```

## 4. Client Directory

**File: `src/actions/clients.ts`**

```typescript
'use server'

import { db } from '@/db'
import { clients } from '@/db/schema'
import { eq, and, ilike, or } from 'drizzle-orm'
import { requireAuth } from '@/lib/get-session'
import { revalidatePath } from 'next/cache'

// ─── Create client ───────────────────────────────────────────────
export async function createClient(formData: {
  name: string
  email: string
  phone?: string
  address?: string
}) {
  const session = await requireAuth()

  if (!formData.name?.trim()) return { error: 'Name is required' }
  if (!formData.email?.trim()) return { error: 'Email is required' }

  const [client] = await db.insert(clients).values({
    userId: session.user.id,
    name: formData.name.trim(),
    email: formData.email.trim(),
    phone: formData.phone?.trim(),
    address: formData.address?.trim(),
  }).returning()

  revalidatePath('/clients')
  return { data: client }
}

// ─── Get all clients ─────────────────────────────────────────────
export async function getClients(search?: string) {
  const session = await requireAuth()

  const results = await db.query.clients.findMany({
    where: search
      ? and(
          eq(clients.userId, session.user.id),
          or(
            ilike(clients.name, `%${search}%`),
            ilike(clients.email, `%${search}%`)
          )
        )
      : eq(clients.userId, session.user.id),
    orderBy: (clients, { desc }) => [desc(clients.createdAt)],
  })

  return { data: results }
}

// ─── Get single client ───────────────────────────────────────────
export async function getClient(id: string) {
  const session = await requireAuth()

  const client = await db.query.clients.findFirst({
    where: and(eq(clients.id, id), eq(clients.userId, session.user.id)),
    with: { invoices: { orderBy: (inv, { desc }) => [desc(inv.createdAt)] } },
  })

  if (!client) return { error: 'Client not found' }
  return { data: client }
}

// ─── Update client ───────────────────────────────────────────────
export async function updateClient(id: string, formData: Partial<{
  name: string; email: string; phone: string; address: string
}>) {
  const session = await requireAuth()

  const existing = await db.query.clients.findFirst({
    where: and(eq(clients.id, id), eq(clients.userId, session.user.id)),
  })
  if (!existing) return { error: 'Client not found' }

  const [updated] = await db.update(clients)
    .set(formData)
    .where(and(eq(clients.id, id), eq(clients.userId, session.user.id)))
    .returning()

  revalidatePath('/clients')
  revalidatePath(`/clients/${id}`)
  return { data: updated }
}

// ─── Delete client ───────────────────────────────────────────────
export async function deleteClient(id: string) {
  const session = await requireAuth()

  // Check for existing invoices first
  const invoiceCount = await db.query.invoices.findFirst({
    where: and(eq(invoices.clientId, id), eq(invoices.userId, session.user.id)),
  })
  if (invoiceCount) {
    return { error: 'Cannot delete a client with existing invoices' }
  }

  await db.delete(clients)
    .where(and(eq(clients.id, id), eq(clients.userId, session.user.id)))

  revalidatePath('/clients')
  return { data: { success: true } }
}
```

**How to call a server action from a component:**

```tsx
// In a Client Component
'use client'
import { createClient } from '@/actions/clients'
import { useTransition } from 'react'

export function NewClientForm() {
  const [isPending, startTransition] = useTransition()

  function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault()
    const form = e.currentTarget
    startTransition(async () => {
      const result = await createClient({
        name: form.name.value,
        email: form.email.value,
      })
      if (result.error) {
        // show error toast
      }
    })
  }

  return (
    <form onSubmit={handleSubmit}>
      <input name="name" required />
      <input name="email" type="email" required />
      <button type="submit" disabled={isPending}>
        {isPending ? 'Saving...' : 'Save Client'}
      </button>
    </form>
  )
}
```

**Senior's note:** Always call `revalidatePath()` after a mutation. Without it, Next.js serves the old cached data even though the DB has changed. Call it for every route that would show the affected data.

## 5. Create & Manage Invoices

**File: `src/actions/invoices.ts`**

```typescript
'use server'

import { db } from '@/db'
import { invoices, lineItems, clients } from '@/db/schema'
import { eq, and, desc } from 'drizzle-orm'
import { requireAuth } from '@/lib/get-session'
import { revalidatePath } from 'next/cache'

// ─── Generate invoice number ─────────────────────────────────────
async function generateInvoiceNumber(userId: string): Promise<string> {
  const count = await db.$count(invoices, eq(invoices.userId, userId))
  return `INV-${String(count + 1).padStart(3, '0')}`
}

// ─── Calculate totals from line items ────────────────────────────
function calculateTotals(
  items: { quantity: number; unitPrice: number }[],
  taxRate: number
) {
  const subtotal = items.reduce((sum, i) => sum + i.quantity * i.unitPrice, 0)
  const taxAmount = subtotal * (taxRate / 100)
  const total = subtotal + taxAmount
  return { subtotal, taxAmount, total }
}

// ─── Create invoice ───────────────────────────────────────────────
export async function createInvoice(formData: {
  clientId: string
  dueDate: string
  notes?: string
  taxRate: number
  lineItems: { description: string; quantity: number; unitPrice: number }[]
}) {
  const session = await requireAuth()

  if (!formData.clientId) return { error: 'Client is required' }
  if (!formData.dueDate)  return { error: 'Due date is required' }
  if (!formData.lineItems?.length) return { error: 'Add at least one line item' }

  // Confirm client belongs to this user
  const client = await db.query.clients.findFirst({
    where: and(
      eq(clients.id, formData.clientId),
      eq(clients.userId, session.user.id)
    ),
  })
  if (!client) return { error: 'Client not found' }

  const invoiceNumber = await generateInvoiceNumber(session.user.id)
  const { subtotal, taxAmount, total } = calculateTotals(
    formData.lineItems,
    formData.taxRate
  )

  // Transaction: create invoice + all line items atomically
  const result = await db.transaction(async (tx) => {
    const [invoice] = await tx.insert(invoices).values({
      userId:        session.user.id,
      clientId:      formData.clientId,
      invoiceNumber,
      dueDate:       new Date(formData.dueDate),
      notes:         formData.notes,
      taxRate:       formData.taxRate,
      subtotal,
      taxAmount,
      total,
    }).returning()

    await tx.insert(lineItems).values(
      formData.lineItems.map(item => ({
        invoiceId:   invoice.id,
        description: item.description,
        quantity:    item.quantity,
        unitPrice:   item.unitPrice,
        amount:      item.quantity * item.unitPrice,
      }))
    )

    return invoice
  })

  revalidatePath('/invoices')
  return { data: result }
}

// ─── Get all invoices (with filters) ─────────────────────────────
export async function getInvoices(filters?: {
  status?: 'draft' | 'sent' | 'paid' | 'overdue' | 'cancelled'
  clientId?: string
}) {
  const session = await requireAuth()

  const conditions = [eq(invoices.userId, session.user.id)]
  if (filters?.status)   conditions.push(eq(invoices.status, filters.status))
  if (filters?.clientId) conditions.push(eq(invoices.clientId, filters.clientId))

  const results = await db.query.invoices.findMany({
    where: and(...conditions),
    with: { client: { columns: { name: true, email: true } } },
    orderBy: [desc(invoices.createdAt)],
  })

  return { data: results }
}

// ─── Get single invoice ───────────────────────────────────────────
export async function getInvoice(id: string) {
  const session = await requireAuth()

  const invoice = await db.query.invoices.findFirst({
    where: and(eq(invoices.id, id), eq(invoices.userId, session.user.id)),
    with: {
      client: true,
      lineItems: true,
    },
  })

  if (!invoice) return { error: 'Invoice not found' }
  return { data: invoice }
}

// ─── Update invoice (DRAFT only) ──────────────────────────────────
export async function updateInvoice(
  id: string,
  formData: {
    dueDate?: string
    notes?: string
    taxRate?: number
    lineItems?: { description: string; quantity: number; unitPrice: number }[]
  }
) {
  const session = await requireAuth()

  const existing = await db.query.invoices.findFirst({
    where: and(eq(invoices.id, id), eq(invoices.userId, session.user.id)),
  })
  if (!existing)              return { error: 'Invoice not found' }
  if (existing.status !== 'draft') return { error: 'Only draft invoices can be edited' }

  const updatedItems = formData.lineItems ?? []
  const { subtotal, taxAmount, total } = calculateTotals(
    updatedItems,
    formData.taxRate ?? existing.taxRate
  )

  await db.transaction(async (tx) => {
    if (formData.lineItems) {
      await tx.delete(lineItems).where(eq(lineItems.invoiceId, id))
      await tx.insert(lineItems).values(
        formData.lineItems.map(item => ({
          invoiceId:   id,
          description: item.description,
          quantity:    item.quantity,
          unitPrice:   item.unitPrice,
          amount:      item.quantity * item.unitPrice,
        }))
      )
    }

    await tx.update(invoices)
      .set({
        dueDate:   formData.dueDate ? new Date(formData.dueDate) : undefined,
        notes:     formData.notes,
        taxRate:   formData.taxRate,
        subtotal,
        taxAmount,
        total,
        updatedAt: new Date(),
      })
      .where(eq(invoices.id, id))
  })

  revalidatePath('/invoices')
  revalidatePath(`/invoices/${id}`)
  return { data: { success: true } }
}

// ─── Delete invoice (DRAFT only) ─────────────────────────────────
export async function deleteInvoice(id: string) {
  const session = await requireAuth()

  const existing = await db.query.invoices.findFirst({
    where: and(eq(invoices.id, id), eq(invoices.userId, session.user.id)),
  })
  if (!existing)                   return { error: 'Invoice not found' }
  if (existing.status !== 'draft') return { error: 'Only draft invoices can be deleted' }

  // lineItems cascade delete via FK
  await db.delete(invoices).where(eq(invoices.id, id))

  revalidatePath('/invoices')
  return { data: { success: true } }
}
```

## 6. Send Invoice to Client

Sending generates a PDF and emails it. This is a server action because it's a mutation (it changes status), but PDF generation is handled via a utility function.

**Add to `src/actions/invoices.ts`:**

```typescript
import { sendInvoiceEmail } from '@/lib/email'
import { generateInvoicePDF } from '@/lib/pdf'

export async function sendInvoice(id: string, customMessage?: string) {
  const session = await requireAuth()

  const invoice = await db.query.invoices.findFirst({
    where: and(eq(invoices.id, id), eq(invoices.userId, session.user.id)),
    with: { client: true, lineItems: true },
  })

  if (!invoice) return { error: 'Invoice not found' }
  if (invoice.status !== 'draft') return { error: 'Invoice has already been sent' }

  // Get business details for the PDF header
  const business = await db.query.businesses.findFirst({
    where: eq(businesses.userId, session.user.id),
  })

  try {
    const pdfBuffer = await generateInvoicePDF({ invoice, business })

    await sendInvoiceEmail({
      to: invoice.client.email,
      clientName: invoice.client.name,
      invoiceNumber: invoice.invoiceNumber,
      total: invoice.total,
      dueDate: invoice.dueDate,
      pdfBuffer,
      customMessage,
    })

    await db.update(invoices)
      .set({ status: 'sent', sentAt: new Date() })
      .where(eq(invoices.id, id))

    revalidatePath('/invoices')
    revalidatePath(`/invoices/${id}`)
    return { data: { success: true } }

  } catch (err) {
    console.error('Failed to send invoice:', err)
    return { error: 'Failed to send invoice. Please try again.' }
  }
}
```

**File: `src/lib/email.ts`**

```typescript
import nodemailer from 'nodemailer'

const transporter = nodemailer.createTransport({
  host:   process.env.SMTP_HOST,
  port:   Number(process.env.SMTP_PORT),
  auth: {
    user: process.env.SMTP_USER,
    pass: process.env.SMTP_PASS,
  },
})

export async function sendInvoiceEmail({
  to, clientName, invoiceNumber, total, dueDate, pdfBuffer, customMessage
}: {
  to: string
  clientName: string
  invoiceNumber: string
  total: number
  dueDate: Date
  pdfBuffer: Buffer
  customMessage?: string
}) {
  await transporter.sendMail({
    from:    `"${process.env.SMTP_FROM_NAME}" <${process.env.SMTP_FROM_EMAIL}>`,
    to,
    subject: `Invoice ${invoiceNumber}`,
    text:    customMessage ?? `Hi ${clientName}, please find your invoice attached.`,
    attachments: [{
      filename:    `${invoiceNumber}.pdf`,
      content:     pdfBuffer,
      contentType: 'application/pdf',
    }],
  })
}
```

**Senior's note:** For production, swap Nodemailer for Resend or Postmark. Nodemailer is fine for development and low volume, but transactional email services give you delivery tracking, bounce handling, and better inbox rates.

## 7. Invoice Status & Overdue Cron

### Status transition rules

| Current | Allowed next |
|---------|-------------|
| draft | → sent, → cancelled |
| sent | → paid, → overdue (auto), → cancelled |
| overdue | → paid, → cancelled |
| paid | — |
| cancelled | — |

### Overdue cron — Route Handler

Next.js doesn't have built-in cron. Use Vercel Cron Jobs or an external scheduler that calls a protected Route Handler.

**File: `src/app/api/cron/mark-overdue/route.ts`**

```typescript
import { NextResponse } from 'next/server'
import { db } from '@/db'
import { invoices } from '@/db/schema'
import { eq, and, lt } from 'drizzle-orm'

export async function GET(request: Request) {
  // Protect this endpoint — only allow calls from your scheduler
  const authHeader = request.headers.get('authorization')
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const updated = await db.update(invoices)
    .set({ status: 'overdue' })
    .where(
      and(
        eq(invoices.status, 'sent'),
        lt(invoices.dueDate, new Date())
      )
    )
    .returning({ id: invoices.id })

  return NextResponse.json({ marked: updated.length })
}
```

```json
      "path": "/api/cron/mark-overdue",
      "schedule": "0 8 * * *"
    }
  ]
}
```

```typescript
export async function markAsPaid(id: string, paidAt?: string) {
  const session = await requireAuth()

  const invoice = await db.query.invoices.findFirst({
    where: and(eq(invoices.id, id), eq(invoices.userId, session.user.id)),
  })
```

## 9. Basic Dashboard

**File: `src/actions/dashboard.ts`**

```typescript
'use server'

import { db } from '@/db'
import { invoices } from '@/db/schema'
import { eq, desc } from 'drizzle-orm'
import { requireAuth } from '@/lib/get-session'
import { sql } from 'drizzle-orm'

export async function getDashboardStats() {
  const session = await requireAuth()
  const userId = session.user.id

  // One query grouped by status — don't do 5 separate queries
  const grouped = await db
    .select({
      status:   invoices.status,
      count:    sql<number>`count(*)::int`,
      sumTotal: sql<number>`coalesce(sum(${invoices.total}), 0)::real`,
    })
    .from(invoices)
    .where(eq(invoices.userId, userId))
    .groupBy(invoices.status)

  // Turn the array into a lookup object
  const byStatus = Object.fromEntries(
    grouped.map(row => [row.status, row])
  )

  const summary = {
    totalInvoiced:   grouped.reduce((s, r) => s + r.sumTotal, 0),
    totalPaid:       byStatus['paid']?.sumTotal ?? 0,
    totalOutstanding: (byStatus['sent']?.sumTotal ?? 0) + (byStatus['overdue']?.sumTotal ?? 0),
    totalOverdue:    byStatus['overdue']?.sumTotal ?? 0,
  }

  const counts = {
    draft:     byStatus['draft']?.count ?? 0,
    sent:      byStatus['sent']?.count ?? 0,
    paid:      byStatus['paid']?.count ?? 0,
    overdue:   byStatus['overdue']?.count ?? 0,
    cancelled: byStatus['cancelled']?.count ?? 0,
  }

  // Last 5 invoices for the activity feed
  const recent = await db.query.invoices.findMany({
    where: eq(invoices.userId, userId),
    with:  { client: { columns: { name: true } } },
    orderBy: [desc(invoices.createdAt)],
    limit: 5,
  })

  return { data: { summary, counts, recent } }
}
```

```typescript
      .where(eq(invoices.id, id))

    revalidatePath(`/invoices/${id}`)
    return { data: { remindersSent: invoice.remindersSent + 1 } }
  } catch {
    return { error: 'Failed to send reminder' }
  }
}
```

```typescript
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const overdueInvoices = await db.query.invoices.findMany({
    where: and(
      eq(invoices.status, 'overdue'),
      lt(invoices.remindersSent, 3)
    ),
    with: { client: true },
  })

  let sent = 0
  for (const invoice of overdueInvoices) {
    try {
      await sendReminderEmail({ ... })
      await db.update(invoices)
        .set({ remindersSent: invoice.remindersSent + 1 })
        .where(eq(invoices.id, invoice.id))
      sent++
    } catch (err) {
      console.error(`Reminder failed for ${invoice.id}:`, err)
      // Continue — don't let one failure stop the rest
    }
  }

  return NextResponse.json({ sent })
}
```

```typescript
      await tx.insert(lineItems).values(
```

## 13. PDF Generation — Route Handler

PDF download needs to stream a binary response — server actions can't do that. Use a Route Handler.

**File: `src/app/api/invoices/[id]/pdf/route.ts`**

```typescript
import { NextResponse } from 'next/server'
import { auth } from '@/lib/auth'
import { headers } from 'next/headers'
import { db } from '@/db'
import { invoices, businesses } from '@/db/schema'
import { and, eq } from 'drizzle-orm'
import { generateInvoicePDF } from '@/lib/pdf'

export async function GET(
  request: Request,
  { params }: { params: { id: string } }
) {
  const session = await auth.api.getSession({ headers: await headers() })
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const invoice = await db.query.invoices.findFirst({
    where: and(
      eq(invoices.id, params.id),
      eq(invoices.userId, session.user.id)
    ),
    with: { client: true, lineItems: true },
  })

  if (!invoice) return NextResponse.json({ error: 'Not found' }, { status: 404 })

  const business = await db.query.businesses.findFirst({
    where: eq(businesses.userId, session.user.id),
  })

  const pdfBuffer = await generateInvoicePDF({ invoice, business })

  return new NextResponse(pdfBuffer, {
    headers: {
      'Content-Type':        'application/pdf',
      'Content-Disposition': `attachment; filename="${invoice.invoiceNumber}.pdf"`,
    },
  })
}
```

```typescript
import PDFDocument from 'pdfkit'

export function generateInvoicePDF({ invoice, business }: {
  invoice: any // use your Invoice type here
  business: any
}): Promise<Buffer> {
  return new Promise((resolve, reject) => {
    const doc = new PDFDocument({ margin: 50 })
    const chunks: Buffer[] = []

    doc.on('data', chunk => chunks.push(chunk))
    doc.on('end', () => resolve(Buffer.concat(chunks)))
    doc.on('error', reject)

    // Header
    doc.fontSize(20).text(business?.name ?? 'Your Business', { align: 'left' })
    doc.fontSize(10).text(business?.address ?? '').moveDown()

    // Invoice meta
    doc.fontSize(16).text(`Invoice ${invoice.invoiceNumber}`, { align: 'right' })
    doc.fontSize(10)
      .text(`Issued: ${invoice.issueDate.toLocaleDateString()}`, { align: 'right' })
      .text(`Due: ${invoice.dueDate.toLocaleDateString()}`, { align: 'right' })
      .moveDown()

    // Client
    doc.fontSize(12).text('Bill To:')
    doc.fontSize(10).text(invoice.client.name).text(invoice.client.email).moveDown()

    // Line items
    invoice.lineItems.forEach((item: any) => {
      doc.text(`${item.description}`)
      doc.text(`${item.quantity} × £${item.unitPrice.toFixed(2)} = £${item.amount.toFixed(2)}`)
    })
    doc.moveDown()

    // Totals
    doc.text(`Subtotal: £${invoice.subtotal.toFixed(2)}`, { align: 'right' })
    doc.text(`Tax (${invoice.taxRate}%): £${invoice.taxAmount.toFixed(2)}`, { align: 'right' })
    doc.fontSize(12).text(`Total: £${invoice.total.toFixed(2)}`, { align: 'right' })

    if (business?.bankDetails) {
      doc.moveDown().fontSize(10).text('Payment details:').text(business.bankDetails)
    }

    doc.end()
  })
}
```

## 15. Error Handling in Server Actions

Server actions can't throw errors to the client the way REST APIs can. Use a consistent return pattern instead.

### The pattern — always return `{ data }` or `{ error }`

```typescript
// Every action returns one of these two shapes
type ActionResult<T> = 
  | { data: T; error?: never }
  | { error: string; data?: never }
```

### Handle errors in the component

```tsx
const result = await someAction()

if (result.error) {
  toast.error(result.error) // use sonner or react-hot-toast
  return
}

// result.data is available and typed
```

### Catch unexpected errors globally

Wrap the body of every action in a try/catch so a DB error doesn't crash with an unhandled exception:

```typescript
export async function createInvoice(formData: ...) {
  try {
    const session = await requireAuth()
    // ... action logic
    return { data: result }
  } catch (err) {
    if (err instanceof Error && err.message === 'UNAUTHORIZED') {
      return { error: 'You must be logged in' }
    }
    console.error('createInvoice error:', err)
    return { error: 'Something went wrong. Please try again.' }
  }
}
```

## 16. Security Checklist

- [ ] `requireAuth()` called at the top of every server action — no exceptions
- [ ] Every DB query scoped to `userId` — a user must never be able to read or modify another user's data
- [ ] `CRON_SECRET` in `.env` — cron route handlers return 401 without it
- [ ] `.env` in `.gitignore` — `DATABASE_URL`, `BETTER_AUTH_SECRET`, SMTP credentials never committed
- [ ] Drizzle parameterised queries used everywhere — no raw string SQL
- [ ] `BETTER_AUTH_SECRET` is a random 32+ character string (run `openssl rand -base64 32`)
- [ ] `NEXT_PUBLIC_APP_URL` set correctly in production — Better Auth needs this for redirects
- [ ] File uploads (logo) validated for type and size before accepting
- [ ] No sensitive data in `NEXT_PUBLIC_` env vars — those are exposed to the browser
